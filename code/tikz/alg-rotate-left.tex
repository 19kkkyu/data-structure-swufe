\begin{algorithm}
\caption{left-rotate(x)}
\KwIn{A node $x$ of a BST}
\tcp{$x.right$ is not null}
$y\gets x.right$ \\
$x.right\gets y.left$ \\
\If{y.left $\neq$ null}{
    $y.left.p\gets x$
}
$y.p\gets x.p$ \\
\uIf{x.p = null}{
    $root\gets y$ \tcp{if x was the root}
}\uElseIf{x = x.p.left} {
    $x.p.left\gets y$ \tcp{if x was a left child}
}\uElse{
    $x.p.right\gets y$ \tcp{if x was a right child}
}
$y.left\gets x$ \\
$x.p\gets y$ \\
\end{algorithm}
