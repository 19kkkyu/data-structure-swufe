\begin{algorithm}
\caption{get(root, key)}
\KwIn{The $root$ of a BST, and a $key$}
\KwOut{The node containing $key$}
$x\gets root$ \\
\While{x $\neq$ null and x.key $\neq$ key}{
    \If{key $<$ x.key}{
        $x\gets x.left$
    }\Else{
        $x\gets x.right$
    }
}
\Return{x}
\end{algorithm}
