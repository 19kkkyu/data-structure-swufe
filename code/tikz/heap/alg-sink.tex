\begin{algorithm}
\caption{sink(k)}
\KwIn{The position $k$ in $qp$}
\While{2 * k $\leq$ N}{
    $j\gets 2*k$ \tcp{left child}
    \If{j $<$ N and \textnormal{less(}j, j+1\textnormal{)}}{
        $j\gets j + 1$
    }
    \tcp{$j$ is the larger child} 
    \If{not \textnormal{less(}k, j\textnormal{)}}{break}
    swap($k$, $j$) \\
    $k\gets j$
}
\end{algorithm}
