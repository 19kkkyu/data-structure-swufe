\begin{algorithm}
\caption{swim(k)}
\KwIn{The position $k$ in $qp$}
\While{k $>$ 1 and \textnormal{less(}k/2, k\textnormal{)}}{
    swap($k/2$, $k$) \\
    $k\gets k/2$
}
\end{algorithm}
