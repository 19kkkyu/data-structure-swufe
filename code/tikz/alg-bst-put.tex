\begin{algorithm}
\caption{put(x, key)}
\KwIn{A node $x$ of a BST, and a $key$}
\KwOut{The new root of the BST originally rooted at $x$}
\If{x = null}{
    \Return{Node(key)}
}
\If{key $<$ x.key}{
    $x.left\gets put(x.left, key)$
} \ElseIf{key $>$ x.key} {
    $x.right\gets put(x.right, key)$
}
\tcp{Ignore "key = x.key" since duplicated keys are not allowed.}
$x.n \gets size(x.left) + size(x.right) + 1$ \\
\Return{x}
\end{algorithm}
