\begin{algorithm}
\caption{removeMin(x)}
\KwIn{The node $x$ of a BST}
\KwOut{The new root of the BST originally rooted at $x$}
\tcp{assume that x $\neq$ null}
\If{x.left = null}{
    \Return{x.right}
}
$x.left\gets removeMin(x.left)$ \\
$x.n\gets size(x.left) + size(x.right) + 1$ \\
\Return{x}
\end{algorithm}
