\usepackage{xcolor}

\begin{document}
\begin{algorithm}
\caption{transplant(u, v)}
\KwIn{Node $u$ and $v$}
\uIf{u.p = null}{
    $root\gets v$
}\uElseIf{u = u.p.left} {
    $u.p.left\gets v$
}\Else{
    $u.p.right\gets v$
}
\If{v $\ne$ null}{
    $v.p\gets u.p$
}
\end{algorithm}
