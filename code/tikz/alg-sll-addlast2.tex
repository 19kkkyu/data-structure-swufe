\begin{algorithm}
\caption{addLast(head, element)}
\KwIn{$head$ of a linked list, and a new $element$.}
\If{isEmpty(head)}{
    $head\gets$ Node$(element)$
}\Else{
    $tail\gets head$ \\
    \While{tail.next $\neq$ null}{
        $tail\gets tail.next$
    }
    $tail.next \gets$ Node$(element)$
}
\end{algorithm}
