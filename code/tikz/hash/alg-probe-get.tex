\begin{algorithm}
\caption{hash-get(k)}
\KwIn{The key $k$}
\KwOut{The associated value of $k$}
$i\gets 0$ \\
\Repeat{keys[q] = null or i = m}{
    $q\gets h(k, i)$ \\
    \If{keys[q] = k}{
        \Return{values[q]}
    }
    $i\gets i + 1$
}
\Return{null}
\end{algorithm}
