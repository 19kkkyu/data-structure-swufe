\begin{algorithm}
\caption{hash-put(k, v)}
\KwIn{The key $k$, and its associated value $v$}
$i\gets 0$ \\
\Repeat{i = m}{
    $q\gets h(k, i)$ \\
    \uIf{keys[q] = null}{
        \tcp{insert a new pair}
        $keys[q]\gets k$ \\
        $values[q]\gets v$ \\
        $n\gets n + 1$ \\
        \Return{}
    }\uElseIf{keys[q] = k} {
        \tcp{update}
        $values[q]\gets v$
    }
    \uElse{
        $i\gets i + 1$
    }
}
\textbf{error} ``hash table overflow"
\end{algorithm}
