\begin{algorithm}
\caption{selectSort(a)}
\KwIn{An unsorted and comparable array $a$.}
$n\gets a.size()$ \\
\For{$i\gets 0$ \KwTo $n$}{
    $min\gets i$ \tcp{index of minimal element}
    \For{$j\gets i+1$ \KwTo $n$}{
        \If{$a[j] < a[min]$}{
            $min\gets j$
        }
    }
    swap $a[i]$ and $a[min]$
}
\end{algorithm}
