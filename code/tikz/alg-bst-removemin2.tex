\KwIn{The $root$ of a BST}
\If{isEmpty()}{throw error}
$x\gets root$ \\
$y\gets null$ \tcp{$y$ is the partent of $x$}
\While{x.left $\neq$ null}{
\tcp{descending till to the leftmost leaf}
$y\gets x$ \\
$x\gets x.left$ 
}
\If{y = null}{
    \tcp{$x$ is $root$}
    $root\gets x.right$
}\Else{
    $y.left\gets x.right$
}
