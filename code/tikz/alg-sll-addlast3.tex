\begin{algorithm}
\caption{addLast(header, element)}
\KwIn{The $header$ of a linked list, and a new $element$.}
$walk\gets header$ \\
\While{walk.next $\neq$ null}{
    $walk\gets walk.next$
}
\tcp{$walk$ is the last node}
$walk.next\gets Node(element)$
\end{algorithm}
