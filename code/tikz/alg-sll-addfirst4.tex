\begin{algorithm}
\caption{addLast(header, trailer, element)}
\KwIn{$header$ and $trailer$ of a linked list, and a new $element$.} 
$walk\gets header$ \\
\While{walk.next $\neq$ trailer}{
    $walk\gets walk.next$
}
\tcp{$walk$ is the second node to last}
