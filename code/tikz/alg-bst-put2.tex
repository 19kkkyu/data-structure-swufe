\begin{algorithm}
\caption{put(root, key)}
\KwIn{The $root$ of a BST, and a $key$}
$z\gets Node(key)$ \\
$x\gets root$ \tcp{node being compared with $z$}
$y\gets null$ \tcp{$y$ will be parent of $z$}
\While{x $\neq$ null}{
    $y\gets x$ \\
    \uIf{key $<$ x.key}{
        $x\gets x.left$
    }\uElseIf{key $>$ x.key}{
        $x\gets x.right$ 
    }\uElse{
        \Return{} \tcp{stop since no duplicated are allowed} 
    }
}
\uIf{y = null}{
    $root\gets z$
}\uElseIf{key $<$ y.key}{
    $y.left\gets z$
}\uElse{
    $y.right\gets z$
}
\end{algorithm}
