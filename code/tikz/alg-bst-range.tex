\begin{algorithm}
\caption{range(x, key)}
\KwIn{A node $x$ of a BST, result candidates $result$, a low-key $low$, and a high-key $high$}
\If{x = null}{\Return{}}
\If{low $<$ x.key}{
    $range(x.left, result, low, high)$
}
\If{low $\leq$ x.key and high $\geq$ x.key} {
    $result.add(x.key)$
}
\If{high $>$ 0}{
    $range(x.right, result, low, high)$ 
}
\end{algorithm}
