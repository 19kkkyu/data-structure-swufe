\begin{algorithm}
\caption{select(x, k)}
\KwIn{A node $x$ of a BST, and rank $k$}
\KwOut{The node in the tree rooted at $x$ containing key of rank $k$}
\If{x = null}{\Return{null}}
$t\gets size(x.left)$\\
\uIf{t $>$ k}{\Return{select(x.left, k)}}
\uElseIf{t $<$ k}{
    \Return{select(x.right, k-t-1)}
}\Else{
    \Return{x}
}
\end{algorithm}
