\begin{algorithm}
\caption{delete(z)}
\KwIn{The to-be-deleted node $z$}
\If{z.left $\neq$ null and z.right $\neq$ null}{
    \tcp{case 2}
    $y\gets min(z.right)$ \tcp{$y$ is $z$'s successor}
    $z.key\gets y.key$ \\
    $z\gets y$
}
\tcp{case 1}
\If{z.left = null}{
    $transplant(z, z.right)$
}\Else{
    $transplant(z, z.left)$
}
\end{algorithm}
