\begin{algorithm}
\caption{floor(x, key)}
\KwIn{A node $x$ of a BST, and a $key$}
\KwOut{The node in the tree rooted at $x$ whose key is the floor of $key$}
\If{x = null}{\Return{null}}
\uIf{key = x.key}{
    \Return{x}
}\uElseIf{key $<$ x.key} {
    \Return{floor(x.left, key)}
}
$t\gets floor(x.right, key)$ \\
\If{t $\neq$ null}{
    \Return{t}
}\Else{
    \Return{x}
}
\end{algorithm}
