\begin{document}
\begin{algorithm}
\caption{size(head)}
\KwIn{Head node $head$ of a linked list.}
\KwOut{The number of elements in this linked list.}
$cnt\gets 0$ \\
$node\gets head$ \\
\While{$node \neq null$}{
    $cnt\gets cnt + 1$ \\
    $node\gets node.next$
}
\Return{cnt}
\end{algorithm}
