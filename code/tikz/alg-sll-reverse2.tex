\begin{algorithm}
\caption{reverseList(head)}
\KwIn{The $head$ of a linked list.}
\KwOut{The reversed $head$ of the linked list.}
\If{head = null or head.next = null}{
    \Return{head}
}
$restHead\gets reverseList(head.next)$ \\
$head.next.next\gets head$ \\
$head.next\gets null$\\
\Return{restHead}
\end{algorithm}
