\begin{algorithm}
\caption{remove(x, key)}
\KwIn{The node $x$ of a BST, and a $key$}
\KwOut{The new root of the BST originally rooted at $x$}
\If{x = null}{
    \Return{null}
}
\uIf{key $<$ x.key}{
    $x.left\gets remove(x.left, key)$
}\uElseIf{key $>$ x.key}{
    $x.right\gets remove(x.right, key)$
}\Else{
    \If{x.right = null}{
        \Return{x.left}
    }
    \If{x.left = null}{
        \Return{x.right}
    }
    $t\gets x$ \\
    $x\gets min(t.right)$ \\
    $x.right\gets removeMin(t.right)$ \\
    $x.left\gets t.left$
}
$x.n\gets size(x.left) + size(x.right) + 1$ \\
\Return{x}
\end{algorithm}
