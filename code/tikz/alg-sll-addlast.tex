\begin{algorithm}
\caption{addLast(head, tail, element)}
\KwIn{$head$ and $tail$ of a linked list, and a new $element$.}
$oldTail\gets tail$ \\
$tail\gets$Node$(element)$ \\
\If{isEmpty($head$)}{
    $head \gets tail$
} \Else{
    $oldTail.next \gets tail$
}
\end{algorithm}
