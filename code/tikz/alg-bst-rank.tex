\begin{algorithm}
\caption{rank(x, key)}
\KwIn{A node $x$ of a BST, and a key $key$}
\KwOut{Number of keys less than $x.key$ in the subtree rooted at $x$}
\If{x = null}{
    \Return{0}
}
\uIf{key $<$ x.key}{
    \Return{rank(x.left, key)}
}\uElseIf{key $>$ x.key}{
    \Return{1 + size(x.left) + rank(x.right, key)}
}\Else{
    \Return{size(x.left)}
}
\end{algorithm}
